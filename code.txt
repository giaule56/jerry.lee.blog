String doublyNotLess(String N) {
  int reverse = 0;
  int n = Integer.valueOf(N);
  
  for (int i = n+1; ; i++){
  	while( i != 0)
    {
      reverse = reverse * 10;
      reverse = reverse + i%10;
      i = i/10;

    }
    if(reverse > n){
    	break;
    }
  }
	
  return String.valueOf(reverse);
}
